<script>
import {reactive, toRefs, defineComponent} from 'vue';

export default defineComponent({
    setup() {
        const data = reactive({
            contentMenuList: {
                menus: [
                    {
                        command: 'delete',
                        name: '删除',
                    },
                    {
                        command: 'person',
                        name: '个人中心个人中心个人中心',
                    },
                    {
                        command: 'logout',
                        name: '退出',
                    },
                ],
                callback(item) {
                    console.warn(item, '鼠标右键菜单点击事件')
                },
            },
            contentMenuListCustom: {
                menus: [
                    {
                        command: 'delete',
                        name: '删除',
                    },
                    {
                        command: 'person',
                        name: '个人中心个人中心个人中心',
                    },
                    {
                        command: 'logout',
                        name: '退出',
                    },
                ],
                callback(item) {
                    console.warn(item, '鼠标右键菜单点击事件')
                },
                hoverStyle: {
                    color: 'white',
                    backgroundColor: '#409EFF',
                },
            }
        })
        return {
            ...toRefs(data)
        }
    },
})
</script>
